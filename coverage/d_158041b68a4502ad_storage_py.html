271 """Push setting that enables timearea behaviour, (see __init__ for

275 Args:

619 def biochemical_processes(self): 

620 """Runs all biochemical processes and updates pollutant amounts 

621 

622 Returns: 

623 in_ (dict): A VQIP amount that represents total gain in pollutant amounts 

624 out_ (dict): A VQIP amount that represents total loss in pollutant amounts 

625 """ 

626 #TODO make more modular 

627 self.update_depth() 

628 

629 self.tank.storage['temperature'] = (1 - 1 / self.T_wdays) * self.tank.storage['temperature'] + (1 / self.T_wdays) * self.get_data_input('temperature') 

630 

631 #Update lagged temperatures 

632 if len(self.lagged_temperatures) > self.max_temp_lag: 

633 del self.lagged_temperatures[0] 

634 self.lagged_temperatures.append(self.tank.storage['temperature']) 

635 

636 #Update lagged total phosphorus 

637 if len(self.lagged_total_phosphorus) > self.max_phosphorus_lag: 

638 del self.lagged_total_phosphorus[0] 

639 total_phosphorus = self.tank.storage['phosphate'] + self.tank.storage['org-phosphorus'] 

640 self.lagged_total_phosphorus.append(total_phosphorus) 

641 

642 #Check if any water 

643 if self.tank.storage['volume'] < constants.FLOAT_ACCURACY: 

644 #Assume these only do something when there is water 

645 return (self.empty_vqip(), self.empty_vqip()) 

646 

647 if self.tank.storage['temperature'] <= 0 : 

648 #Seems that these things are only active when above freezing 

649 return (self.empty_vqip(), self.empty_vqip()) 

650 

651 #Denitrification 

652 tempfcn = 2 ** ((self.tank.storage['temperature'] - 20) / 10) 

653 if self.tank.storage['temperature'] < 5 : 

654 tempfcn *= (self.tank.storage['temperature'] / 5) 

655 

656 din = self.get_din_pool() 

657 din_concentration = din / self.tank.storage['volume'] 

658 confcn = din_concentration / (din_concentration + self.halfsatINwater) # [kg/m3] 

659 denitri_water = self.denpar_w * self.area * tempfcn * confcn # [kg/day] #TODO convert to per DT 

660 

661 river_denitrification = min(denitri_water, 0.5 * din) # [kg/day] max 50% kan be denitrified 

662 din_ = (din - river_denitrification) # [kg] 

663 

664 #Update mass balance 

665 in_ = self.empty_vqip() 

666 out_ = self.empty_vqip() 

667 if din > 0: 

668 for pol in self.din_components: 

669 #denitrification 

670 loss = (din - din_) / din * self.tank.storage[pol] 

671 out_[pol] += loss 

672 self.tank.storage[pol] -= loss 

673 

674 din = self.get_din_pool() 

675 

676 #Calculate moving averages  

677 #TODO generalise 

678 temp_10_day = sum(self.lagged_temperatures[-10:]) / 10 

679 temp_20_day = sum(self.lagged_temperatures[-20:]) / 20 

680 total_phos_365_day = sum(self.lagged_total_phosphorus) / self.max_phosphorus_lag 

681 

682 #Calculate coefficients 

683 tempfcn = (self.tank.storage['temperature']) / 20 * (temp_10_day - temp_20_day) / 5 

684 if (total_phos_365_day - self.limpppar + self.hsatTP) > 0: 

685 totalphosfcn = (total_phos_365_day - self.limpppar) / (total_phos_365_day - self.limpppar + self.hsatTP) 

686 else: 

687 totalphosfcn = 0 

688 

689 #Mineralisation/production 

690 #TODO this feels like it could be much tidier 

691 minprodN = self.prodNpar * totalphosfcn * tempfcn * self.area * self.current_depth # [kg N/day] 

692 minprodP = self.prodPpar * totalphosfcn * tempfcn * self.area * self.current_depth * self.uptake_PNratio # [kg N/day] 

693 if minprodN > 0 : 

694 #production (inorg -> org) 

695 minprodN = min(0.5 * din, minprodN) # only half pool can be used for production 

696 minprodP = min(0.5 * self.tank.storage['phosphate'], minprodP) # only half pool can be used for production 

697 

698 #Update mass balance 

699 out_['phosphate'] = minprodP 

700 self.tank.storage['phosphate'] -= minprodP 

701 in_['org-phosphorus'] = minprodP 

702 self.tank.storage['org-phosphorus'] += minprodP 

703 if din > 0: 

704 for pol in self.din_components: 

705 loss = minprodN * self.tank.storage[pol] / din 

706 out_[pol] += loss 

707 self.tank.storage[pol] -= loss 

708 

709 in_['org-nitrogen'] = minprodN 

710 self.tank.storage['org-nitrogen'] += minprodN 

711 

712 else: 

713 #mineralisation (org -> inorg) 

714 minprodN = min(0.5 * self.tank.storage['org-nitrogen'], -minprodN) 

715 minprodP = min(0.5 * self.tank.storage['org-phosphorus'], -minprodP) 

716 

717 #Update mass balance 

718 in_['phosphate'] = minprodP 

719 self.tank.storage['phosphate'] += minprodP 

720 out_['org-phosphorus'] = minprodP 

721 self.tank.storage['org-phosphorus'] -= minprodP 

722 if din > 0: 

723 for pol in self.din_components: 

724 gain = minprodN * self.tank.storage[pol] / din 

725 in_[pol] += gain 

726 self.tank.storage[pol] += gain 

727 

728 out_['org-nitrogen'] = minprodN 

729 self.tank.storage['org-nitrogen'] -= minprodN 

730 

731 din = self.get_din_pool() 

732 

733 # macrophyte uptake 

734 # temperature dependence factor 

735 tempfcn1 = (max(0, self.tank.storage['temperature']) / 20) ** 0.3 

736 tempfcn2 = (self.tank.storage['temperature'] - temp_20_day) / 5 

737 tempfcn = max(0, tempfcn1 * tempfcn2) 

738 

739 macrouptN = self.muptNpar * tempfcn * self.area # [kg/day] 

740 macrophyte_uptake_N = min(0.5 * din, macrouptN) 

741 if din > 0: 

742 for pol in self.din_components: 

743 loss = macrophyte_uptake_N * self.tank.storage[pol] / din 

744 out_[pol] += loss 

745 self.tank.storage[pol] -= loss 

746 

747 macrouptP = self.muptPpar * tempfcn * max(0, totalphosfcn) * self.area # [kg/day] 

748 macrophyte_uptake_P = min(0.5 * self.tank.storage['phosphate'], macrouptP) 

749 out_['phosphate'] += macrophyte_uptake_P 

750 self.tank.storage['phosphate'] -= macrophyte_uptake_P 

751 

752 #TODO 

753 #source/sink for benthos sediment P 

754 #suspension/resuspension 

755 return in_, out_

757 def get_riverrc(self): 

758 """Get river outflow coefficient (i.e., how much water leaves the tank in this  

759 timestep) 

760 

761 Returns: 

762 riverrc (float): outflow coeffficient 

763 """ 

764 #Calculate travel time 

765 total_time = self.length / self.velocity 

766 #Apply damp 

767 kt = self.damp * total_time # [day] 

768 if kt != 0 : 

769 riverrc = 1 - kt + kt * exp(-1 / kt) # [-] 

770 else: 

771 riverrc = 1 

772 return riverrc

899 vqip (dict): A VQIP amount to be received

901 Returns: 

902 reply (dict): A VQIP amount that was not successfully received 

903 """ 

904 #Apply normal reservoir storage 

905 #We do this under the assumption that spill is mixed in with the reservoir 

906 #If the reservoir can't spill everything you'll get some weird numbers in 

907 #reply, but if your reservoir can't spill as much as you like then you

922 def push_check_river_reservoir(self, vqip = None):

928 the return value (only volume key is used). Defaults to None. 

929

931 excess (dict): A VQIP amount of water that cannot be received 

932 """

935 of_type = ['Node','River','Waste'] 

936 )['avail']

950 """Satisfy environmental flow downstream 

951 """
